145 [[nodiscard]] mapped_type get(const key_type& key) const {
146 if (!dirty_table_[key] && cache_.contains(key)) {
147 return cache_.get(key);
148 }
149
150 if (c0_layer_.contains(key)) {
151 const auto value{c0_layer_.get(key)};
152 cache_.put(key, value);
153 dirty_table_[key] = false;
154 return value;
155 }
156
157 for (const auto& ck_layer : ck_layers_) {
158 for (const auto& sstable : ck_layer | std::views::reverse) {
159 if (sstable.contains(key)) {
160 const auto value{sstable.get(key)};
161 cache_.put(key, value);
162 dirty_table_[key] = false;
163 return value;
164 }
165 }
166 }
167
168 return std::nullopt;

181 [[nodiscard]] std::vector<value_type> getRange(
182 const key_type& start,
183 const key_type& end,
184 TimeSeriesKeyFilter&& filter
185 ) const {
186 table_type entry_table;
187 for (const auto& ck_layer : ck_layers_ | std::views::reverse) {
188 for (const auto& sstable : ck_layer) {
189 for (const auto& [key, value] : sstable.getRange(start, end)) {
190 if (!filter(key)) {
191 continue;
192 }
193 if (!value.has_value()) {
194 entry_table.erase(key);
195 continue;
196 }
197 entry_table[key] = value;
198 }
199 }
200 }
201 for (const auto& [key, value] : c0_layer_.getRange(start, end)) {
202 if (!filter(key)) {
203 continue;
204 }
205 if (!value.has_value()) {
206 entry_table.erase(key);
207 continue;
208 }
209 entry_table[key] = value;
210 }
211 return {entry_table.begin(), entry_table.end()};
212 }

419 void load_sstables() {
420 for (size_type k{1}; k < ck_layers_.size(); ++k) {
421 ck_layers_[k].reserve(CK_LAYER_TABLE_COUNT[k]);
422 }
423 auto& c1_layer{ck_layers_[1]};
424 std::array<std::set<FilePath>, LAYER_COUNT> sstable_files;
425 for (const auto& file : std::filesystem::directory_iterator(path_)) {
426 if (!file.is_regular_file() || file.path().extension() != ".sst") {
427 continue;
428 }
429 const auto l_pos{file.path().filename().string().find("_l")};
430 const auto layer_idx{
431 std::stoull(file.path().filename().string().substr(l_pos + 2, 1))
432 };
433 sstable_files[layer_idx].insert(file.path());
434 }
435 for (size_type k{1}; k < ck_layers_.size(); ++k) {
436 for (const auto& sstable_file : sstable_files[k]) {
437 ck_layers_[k].emplace_back(sstable_file);
438 }
439 }
440 compact();
441 }

528 void compact_layer(size_type k) {
529 auto& curr_layer{ck_layers_[k]};
530 auto& next_layer{ck_layers_[k + 1]};
531 const auto window_size{CK_LAYER_WINDOW_SIZE[k + 1]};
532
533 std::map<TimeWindow, table_type, std::greater<>> time_window_to_entries;
534 std::vector<FilePath> files_to_remove;
535
536 for (auto& sstable : curr_layer) {
537 for (const auto& [key, value] : sstable.getRange(
538 MIN_TIME_SERIES_KEY,
539 MAX_TIME_SERIES_KEY
540 )) {
541 const auto start{(key.timestamp() / window_size) * window_size};
542 const auto time_window{TimeWindow{start, start + window_size}};
543 time_window_to_entries[time_window][key] = value;
544 }
545 files_to_remove.push_back(sstable.path());
546 files_to_remove.push_back(sstable.metadataPath());
547 }
548
549 reset_layer(k);
550
551 for (auto& [time_window, entries] : time_window_to_entries) {
552 auto sstable{merge_entries(std::move(entries), k)};
553 next_layer.push_back(std::move(sstable));
554 }
555
556 for (const auto& file : files_to_remove) {
557 if (!std::filesystem::remove(file)) {
558 throw std::runtime_error{
559 "LSMTree::compact_layer(): Failed to remove file: '"
560 + std::string{file} + "' after compaction."
561 };
562 }
563 }
564 }

573 SSTable<TValue> merge_entries(table_type&& entries, size_type k) {
574 auto memtable{MemTable<TValue>{std::move(entries)}};
575 auto sstable{SSTable<TValue>{
576 get_next_file_path(k + 1),
577 std::move(memtable),

579 }};
580 return sstable;
