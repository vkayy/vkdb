145 [[nodiscard]] mapped_type get(const key_type& key) const {
146 if (!dirty_table_[key] && cache_.contains(key)) {
147 return cache_.get(key);
148 }
149
150 if (c0_layer_.contains(key)) {
151 const auto value{c0_layer_.get(key)};
152 cache_.put(key, value);
153 dirty_table_[key] = false;
154 return value;
155 }
156
157 for (const auto& ck_layer : ck_layers_) {
158 for (const auto& sstable : ck_layer | std::views::reverse) {
159 if (sstable.contains(key)) {
160 const auto value{sstable.get(key)};
161 cache_.put(key, value);
162 dirty_table_[key] = false;
163 return value;
164 }
165 }
166 }
167
168 return std::nullopt;

181 [[nodiscard]] std::vector<value_type> getRange(
182 const key_type& start,
183 const key_type& end,
184 TimeSeriesKeyFilter&& filter
185 ) const {
186 table_type entry_table;
187 for (const auto& ck_layer : ck_layers_ | std::views::reverse) {
188 for (const auto& sstable : ck_layer) {
189 for (const auto& [key, value] : sstable.getRange(start, end)) {
190 if (!filter(key)) {
191 continue;
192 }
193 if (!value.has_value()) {
194 entry_table.erase(key);
195 continue;
196 }
197 entry_table[key] = value;
198 }
199 }
200 }
201 for (const auto& [key, value] : c0_layer_.getRange(start, end)) {
202 if (!filter(key)) {
203 continue;
204 }
205 if (!value.has_value()) {
206 entry_table.erase(key);
207 continue;
208 }
209 entry_table[key] = value;
210 }
211 return {entry_table.begin(), entry_table.end()};
212 }

423 void load_sstables() {
424 for (size_type k{1}; k < ck_layers_.size(); ++k) {
425 ck_layers_[k].reserve(CK_LAYER_TABLE_COUNT[k]);
426 }
427 auto& c1_layer{ck_layers_[1]};
428 std::array<std::set<FilePath>, LAYER_COUNT> sstable_files;
429 for (const auto& file : std::filesystem::directory_iterator(path_)) {
430 if (!file.is_regular_file() || file.path().extension() != ".sst") {
431 continue;
432 }
433 const auto l_pos{file.path().filename().string().find("_l")};
434 const auto layer_idx{
435 std::stoull(file.path().filename().string().substr(l_pos + 2, 1))
436 };
437 sstable_files[layer_idx].insert(file.path());
438 }
439 for (size_type k{1}; k < ck_layers_.size(); ++k) {
440 for (const auto& sstable_file : sstable_files[k]) {
441 ck_layers_[k].emplace_back(sstable_file);
442 }
443 }
444 compact();

532 void compact_layer(size_type k) {
533 auto& curr_layer{ck_layers_[k]};
534 auto& next_layer{ck_layers_[k + 1]};
535 const auto window_size{CK_LAYER_WINDOW_SIZE[k + 1]};
536
537 std::map<TimeWindow, table_type, std::greater<>> time_window_to_entries;
538 std::vector<FilePath> files_to_remove;
539
540 for (auto& sstable : curr_layer) {
541 for (const auto& [key, value] : sstable.getRange(
542 MIN_TIME_SERIES_KEY,
543 MAX_TIME_SERIES_KEY
544 )) {
545 const auto start{(key.timestamp() / window_size) * window_size};
546 const auto time_window{TimeWindow{start, start + window_size}};
547 time_window_to_entries[time_window][key] = value;
548 }
549 files_to_remove.push_back(sstable.path());
550 files_to_remove.push_back(sstable.metadataPath());
551 }
552
553 reset_layer(k);
554
555 for (auto& [time_window, entries] : time_window_to_entries) {
556 auto sstable{merge_entries(std::move(entries), k)};
557 next_layer.push_back(std::move(sstable));
558 }
559
560 for (const auto& file : files_to_remove) {
561 if (!std::filesystem::remove(file)) {
562 throw std::runtime_error{
563 "LSMTree::compact_layer(): Failed to remove file: '"
564 + std::string{file} + "' after compaction."
565 };
566 }
567 }
568 }

577 SSTable<TValue> merge_entries(table_type&& entries, size_type k) {
578 auto memtable{MemTable<TValue>{std::move(entries)}};
579 auto sstable{SSTable<TValue>{
580 get_next_file_path(k + 1),
581 std::move(memtable),
582 memtable.size()
583 }};
584 return sstable;
