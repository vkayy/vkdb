145 [[nodiscard]] mapped_type get(const key_type& key) const {
146 if (!dirty_table_[key] && cache_.contains(key)) {
147 return cache_.get(key);
148 }
149
150 if (mem_table_.contains(key)) {
151 const auto value{mem_table_.get(key)};
152 cache_.put(key, value);
153 dirty_table_[key] = false;
154 return value;
155 }
156
157 for (const auto& ck_layer : ck_layers_) {
158 for (const auto& sstable : ck_layer | std::views::reverse) {
159 if (sstable.contains(key)) {
160 const auto value{sstable.get(key)};
161 cache_.put(key, value);
162 dirty_table_[key] = false;
163 return value;
164 }
165 }
166 }
167
168 return std::nullopt;

181 [[nodiscard]] std::vector<value_type> getRange(
182 const key_type& start,
183 const key_type& end,
184 TimeSeriesKeyFilter&& filter
185 ) const {
186 table_type entry_table;
187 for (const auto& ck_layer : ck_layers_ | std::views::reverse) {
188 for (const auto& sstable : ck_layer) {
189 for (const auto& [key, value] : sstable.getRange(start, end)) {
190 if (!filter(key)) {
191 continue;
192 }
193 if (!value.has_value()) {
194 entry_table.erase(key);
195 continue;
196 }
197 entry_table[key] = value;
198 }
199 }
200 }
201 for (const auto& [key, value] : mem_table_.getRange(start, end)) {
202 if (!filter(key)) {
203 continue;
204 }
205 if (!value.has_value()) {
206 entry_table.erase(key);
207 continue;
208 }
209 entry_table[key] = value;
210 }
211 return {entry_table.begin(), entry_table.end()};
212 }

406 void load_sstables() {
407 std::array<std::set<FilePath>, LAYER_COUNT> sstable_files;
408 for (const auto& file : std::filesystem::directory_iterator(path_)) {
409 if (!file.is_regular_file() || file.path().extension() != ".sst") {
410 continue;
411 }
412 const auto l_pos{file.path().filename().string().find("_l")};
413 const auto layer_idx{
414 std::stoull(file.path().filename().string().substr(l_pos + 2))
415 };
416 sstable_files[layer_idx].insert(file.path());
417 const auto id{
418 std::stoull(file.path().filename().string().substr(l_pos + 4))
419 };
420 sstable_id_[layer_idx] = std::max(sstable_id_[layer_idx], id + 1);
421 }
422 for (size_type k{0}; k < LAYER_COUNT; ++k) {
423 for (const auto& sstable_file : sstable_files[k]) {
424 ck_layers_[k].emplace_back(std::move(sstable_file));
425 }
426 }
427 compact();
428 }

602 void merge_oldest_in_layer(size_type k) {
603 auto& curr_layer{ck_layers_[k]};
604 auto& next_layer{ck_layers_[k + 1]};
605 const auto window_size{CK_LAYER_WINDOW_SIZE[k + 1]};
606
607 TimeWindowToEntriesMap time_window_to_entries;
608 std::vector<FilePath> files_to_remove;

612 sstable, window_size,
613 time_window_to_entries,
614 files_to_remove

623 time_window_to_entries,
624 files_to_remove

630 if (!std::filesystem::remove(file)) {

633 + std::string{file} + "' after compaction."
634 };
635 }
636 }
637
638 for (auto& [time_window, entries] : time_window_to_entries) {
639 auto sstable{merge_entries(std::move(entries), k)};

651 SSTable<TValue> merge_entries(table_type&& entries, size_type k) {
652 auto memtable{MemTable<TValue>{std::move(entries)}};
653 auto sstable{SSTable<TValue>{

655 std::move(memtable),
