186 mapped_type binary_layer_search(
187 size_type k,
188 const key_type& key
189 ) const {
190 const auto timestamp{key.timestamp()};
191 const auto& ck_layer{ck_layers_[k]};
192 auto sstable_it{std::lower_bound(
193 ck_layer.begin(),
194 ck_layer.end(),
195 timestamp,
196 [](const auto& sstable, const auto target_time) {
197 return target_time < sstable.timeRange().lower();
198 }
199 )};
200
201 if (
202 sstable_it != ck_layer.end() &&
203 sstable_it->timeRange().lower() <= timestamp &&
204 timestamp <= sstable_it->timeRange().upper()
205 ) {
206 const auto sstable_value{sstable_it->get(key)};
207 if (sstable_value.has_value()) {
208 cache_.put(key, sstable_value);
209 dirty_table_[key] = false;
210 return sstable_value;
211 }
212 return std::nullopt;
213 }
214 return std::nullopt;
215 }

253 }

273 [[nodiscard]] mapped_type get(const key_type& key) const noexcept {
274 if (!dirty_table_[key] && cache_.contains(key)) {
275 return cache_.get(key);
276 }
277
278 try {
279 return search_memtable(key);
280 } catch (const std::invalid_argument& e) {
281 const auto c0_value{iterative_layer_search(0, key)};
282 if (c0_value.has_value()) {
283 return c0_value;
284 }
285
286 for (size_type k{1}; k < LAYER_COUNT; ++k) {
287 const auto ck_value{binary_layer_search(k, key)};
288 if (ck_value.has_value()) {
289 return ck_value;
290 }
291 }
292
293 return std::nullopt;
294 }

519 void load_sstables() {
520 std::array<std::set<FilePath>, LAYER_COUNT> sstable_files;
521 for (const auto& file : std::filesystem::directory_iterator(path_)) {
522 if (!file.is_regular_file() || file.path().extension() != ".sst") {
523 continue;
524 }
525 const auto l_pos{file.path().filename().string().find("_l")};
526 const auto layer_idx{
527 std::stoull(file.path().filename().string().substr(l_pos + 2))
528 };
529 sstable_files[layer_idx].insert(file.path());
530 const auto id{
531 std::stoull(file.path().filename().string().substr(l_pos + 4))
532 };
533 sstable_id_[layer_idx] = std::max(sstable_id_[layer_idx], id + 1);
534 }
535 for (size_type k{0}; k < LAYER_COUNT; ++k) {
536 for (const auto& sstable_file : sstable_files[k]) {
537 ck_layers_[k].emplace_back(std::move(sstable_file));
538 }
539 }
540 compact();
541 }

689 void update_entries_with_sstable(
690 SSTable<TValue>&& sstable,
691 size_type window_size,
692 TimeWindowToEntriesMap& time_window_to_entries,
693 std::vector<FilePath>& files_to_remove
694 ) const noexcept {
695 for (const auto& [key, value] : sstable.entries()) {
696 const auto start{(key.timestamp() / window_size) * window_size};
697 const auto time_window{TimeWindow{start, start + window_size}};
698 time_window_to_entries[time_window][key] = value;

700 files_to_remove.push_back(sstable.path());
701 files_to_remove.push_back(sstable.metadataPath());

709 void merge_all_in_layer(size_type k) {
710 auto& curr_layer{ck_layers_[k]};
711 auto& next_layer{ck_layers_[k + 1]};
712 const auto window_size{CK_LAYER_WINDOW_SIZE[k + 1]};
713
714 TimeWindowToEntriesMap time_window_to_entries;
715 std::vector<FilePath> files_to_remove;

718 update_entries_with_sstable(

720 time_window_to_entries,
721 files_to_remove

757 void merge_oldest_in_layer(size_type k) {
758 auto& curr_layer{ck_layers_[k]};
759 auto& next_layer{ck_layers_[k + 1]};
760 const auto window_size{CK_LAYER_WINDOW_SIZE[k + 1]};
761
762 TimeWindowToEntriesMap time_window_to_entries;
763 std::vector<FilePath> files_to_remove;
764
765 for (auto&& sstable : next_layer) {
766 update_entries_with_sstable(
767 std::move(sstable), window_size,
768 time_window_to_entries,
769 files_to_remove
770 );
771 }
772 reset_layer(k + 1);
773

775 auto& oldest_sstable{curr_layer.front()};
776 update_entries_with_sstable(
777 std::move(oldest_sstable), window_size,
778 time_window_to_entries,
779 files_to_remove
780 );
781 curr_layer.pop_front();
782 }
783
784 for (const auto& file : files_to_remove) {
785 if (!std::filesystem::remove(file)) {
786 throw std::runtime_error{
787 "LSMTree::merge_oldest_in_layer(): Failed to remove file: '"
788 + std::string{file} + "' after compaction."
789 };
790 }
791 }
792
793 for (auto& [time_window, entries] : time_window_to_entries) {
794 auto sstable{merge_entries(std::move(entries), k)};
795 next_layer.push_back(std::move(sstable));
796 }
