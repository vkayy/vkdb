148 [[nodiscard]] mapped_type get(const key_type& key) const {
149 if (!dirty_table_[key] && cache_.contains(key)) {
150 return cache_.get(key);
151 }
152 if (mem_table_.contains(key)) {
153 const auto value{mem_table_.get(key)};
154 cache_.put(key, value);
155 dirty_table_[key] = false;
156 return value;
157 }
158 for (const auto& sstable : sstables_ | std::views::reverse) {
159 if (sstable.contains(key)) {
160 const auto value{sstable.get(key)};
161 cache_.put(key, value);

179 [[nodiscard]] std::vector<value_type> getRange(
180 const key_type& start,
181 const key_type& end,
182 TimeSeriesKeyFilter&& filter
183 ) const {
184 table_type entry_table;
185 for (const auto& sstable : sstables_) {
186 for (const auto& [key, value] : sstable.getRange(start, end)) {
187 if (!filter(key)) {
188 continue;
189 }
190 if (!value.has_value()) {
191 entry_table.erase(key);
192 continue;
193 }
194 entry_table[key] = value;
195 }
196 }
197 for (const auto& [key, value] : mem_table_.getRange(start, end)) {
198 if (!filter(key)) {
199 continue;
200 }
201 if (!value.has_value()) {
202 entry_table.erase(key);
203 continue;
204 }
205 entry_table[key] = value;
206 }
207 return {entry_table.begin(), entry_table.end()};
208 }

220 [[nodiscard]] std::vector<value_type> getRangeParallel(
221 const key_type& start,
222 const key_type& end,
223 TimeSeriesKeyFilter filter
224 ) const {
225 std::vector<std::future<std::vector<value_type>>> range_futures;
226 range_futures.reserve(sstables_.size() + 1);
227
228 range_futures.push_back(std::async(std::launch::async,
229 [this, &start, &end, &filter]() {
230 std::vector<value_type> mem_table_entries;
231 for (const auto& [key, value] : mem_table_.getRange(start, end)) {
232 if (filter(key)) {
233 mem_table_entries.emplace_back(key, value);
234 }
235 }
236 return mem_table_entries;
237 }
238 ));
239
240 for (const auto& sstable : sstables_ | std::views::reverse) {
241 range_futures.push_back(std::async(std::launch::async,
242 [&sstable, &start, &end, &filter]() {
243 std::vector<value_type> sstable_entries;
244 for (const auto& [key, value] : sstable.getRange(start, end)) {
245 if (filter(key)) {
246 sstable_entries.emplace_back(key, value);
247 }
248 }
249 return sstable_entries;
250 }
251 ));
252 }
253
254 table_type entry_table;
255 for (auto& range_future : range_futures) {
256 const auto entries{range_future.get()};
257 for (const auto& [key, value] : entries) {
258 if (!entry_table.contains(key)) {
259 entry_table[key] = value;
260 }
261 }
262 }
263
264 std::vector<value_type> entries;
265 entries.reserve(entry_table.size());
266 for (const auto& [key, value] : entry_table) {
267 if (value.has_value()) {
268 entries.emplace_back(key, value);
269 }
270 }
271
272 return entries;
273 }

386 }

388 sstables_.emplace_back(sstable_file);
