11template <ArithmeticNoCVRefQuals TValue>
12TimeSeriesEntry<TValue> entryFromString(std::string&& entry) {
13 auto sep{entry.find('|')};
14 auto end{entry.find(']')};
15 auto key_str{entry.substr(0, sep)};
16 auto value_str{entry.substr(sep + 1, end - sep - 1)};
17
18 auto entry_key{TimeSeriesKey{std::move(key_str)}};
19 std::optional<TValue> entry_value;
20 if (value_str != "null") {
21 std::stringstream value_ss{value_str};
22 TValue value;
23 value_ss >> value;
24 entry_value = value;
25 }
26
27 return {entry_key, entry_value};
28}
