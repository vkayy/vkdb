389 [[nodiscard]]
390 result_type execute() {
391 switch (query_type_) {
392 case QueryType::NONE:
393 if (filters_.size() == 1) {
394 throw std::runtime_error{
395 "QueryBuilder::execute(): No query type specified."
396 };
397 }
398 set_default_range_if_none();
399 return get_filtered_range();
400 case QueryType::POINT:
401 return execute_point_query();
402 case QueryType::RANGE:
403 return execute_range_query();
404 case QueryType::PUT:
405 return execute_put_query();
406 case QueryType::REMOVE:
407 return execute_remove_query();

565 [[nodiscard]] result_type get_filtered_range() const {
566 if (query_type_ == QueryType::POINT) {
567 return execute_point_query();
568 }
569 const auto& params{std::get<RangeParams>(query_params_)};
570 return lsm_tree_.getRange(params.start, params.end,
571 [this](const auto& k) {
572 return std::ranges::all_of(filters_, [&k](const auto& filter) {
573 return filter(k);
574 });
575 });
576 }

604 [[nodiscard]] result_type execute_point_query() const {
605 const auto& params{std::get<PointParams>(query_params_)};
606 auto value{lsm_tree_.get(params.key)};
607 if (!value.has_value()) {
