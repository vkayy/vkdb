162 [[nodiscard]] mapped_type get(const key_type& key) const {
163 if (!contains(key)) {
164 return std::nullopt;
165 }
166
167 auto pos{index_.at(key)};
168 if (pos >= mmap_size_) {
169 throw std::runtime_error{
170 "SSTable::get(): Invalid position " + std::to_string(pos)
171 + " for key '" + key.str() + "'."
172 };
173 }
174
175 auto entry_ptr{static_cast<const char*>(mmap_) + pos + 1};
176 auto [entry_key, entry_value]
177 = entryFromString<TValue>(std::string{entry_ptr});
178 if (entry_key != key) {
179 throw std::runtime_error{
180 "SSTable::get(): Key mismatch. Expected '" + key.str()
181 + "' but got '" + entry_key.str() + "'."
182 };
183 }
184
185 return entry_value;
186 }

197 [[nodiscard]] std::vector<value_type> getRange(
198 const key_type& start,
199 const key_type& end
200 ) const {
201 if (!overlaps_with(start, end)) {
202 return {};
203 }
204 std::vector<value_type> entries;
205 for (auto it{index_.lower_bound(start)};
206 it != index_.end() && it->first < end; ++it) {
207 const auto [key, pos] = *it;
208 if (key < start) {
209 continue;
210 }
211 if (key > end) {
212 break;
213 }
214 entries.emplace_back(key, get(key));
215 }
216 return entries;
217 }

281 void save_memtable(MemTable<TValue>&& mem_table) {
282 std::ofstream file{file_path_};

288 }

290 file << mem_table.size();
291 for (const auto& [key, value] : mem_table.table()) {

298 }
299 update_metadata(key, pos);

301 }

337 void load_metadata() {
338 std::ifstream file{metadataPath()};
339 if (!file.is_open()) {
340 throw std::runtime_error{
341 "SSTable::load_metadata(): Unable to open file '"
342 + std::string(metadataPath()) + "'."
343 };
344 }
345
346 std::string line;
347 std::getline(file, line);
348 time_range_ = DataRange<Timestamp>{std::move(line)};
349 std::getline(file, line);
350 key_range_ = DataRange<key_type>{std::move(line)};
351 std::getline(file, line);
352 bloom_filter_ = BloomFilter{std::move(line)};
353 std::getline(file, line);
354 auto no_of_entries{std::stoull(line)};
355 for (auto i{0}; i < no_of_entries; ++i) {
356 std::getline(file, line);
357 auto caret_pos{line.find('^')};
358 if (caret_pos == std::string::npos) {
359 throw std::runtime_error{
360 "SSTable::load_metadata(): Invalid index entry '" + line + "'."
361 };
362 }
363 key_type key{std::move(line.substr(0, caret_pos))};
364 auto pos{std::stoull(line.substr(caret_pos + 1))};
365 index_.emplace(key, pos);
366 }
367
368 file.close();
369 }

426 void map_file() {
427 if (!std::filesystem::exists(file_path_)) {
428 std::ofstream file{file_path_, std::ios::app};
429 if (!file.is_open()) {
430 throw std::runtime_error{
431 "SSTable::map_file(): Unable to create file '"
432 + std::string(file_path_) + "'."
433 };
434 }
435 file.close();
436 }
437
438 fd_ = open(file_path_.c_str(), O_RDONLY);
439 if (fd_ == -1) {
440 throw std::runtime_error{
441 "SSTable::map_file(): Unable to open file '"
442 + std::string(file_path_) + "'."
443 };
444 }
445
446 mmap_size_ = std::filesystem::file_size(file_path_);
447
448 mmap_ = mmap(nullptr, mmap_size_, PROT_READ, MAP_PRIVATE, fd_, 0);
449 if (mmap_ == MAP_FAILED) {
450 close(fd_);
451 throw std::runtime_error{
452 "SSTable::map_file(): Unable to map file '"
453 + std::string(file_path_) + "'."
454 };
455 }
456 }

463 void unmap_file() {
464 if (mmap_ != nullptr) {
465 munmap(mmap_, mmap_size_);
466 mmap_ = nullptr;
467 }
468 if (fd_ != -1) {
469 close(fd_);
470 fd_ = -1;
471 }
