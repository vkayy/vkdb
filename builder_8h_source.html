391 result_type execute() {
392 switch (query_type_) {
393 case QueryType::NONE:
394 if (filters_.size() == 1) {
395 throw std::runtime_error{
396 "QueryBuilder::execute(): No query type specified."
397 };
398 }
399 set_default_range_if_none();
400 return get_filtered_range();
401 case QueryType::POINT:
402 return execute_point_query();
403 case QueryType::RANGE:
404 return execute_range_query();
405 case QueryType::PUT:
406 return execute_put_query();
407 case QueryType::REMOVE:

566 [[nodiscard]] result_type get_filtered_range() const {
567 if (query_type_ == QueryType::POINT) {
568 return execute_point_query();
569 }
570 const auto& params{std::get<RangeParams>(query_params_)};
571 return lsm_tree_.getRange(params.start, params.end,
572 [this](const auto& k) {
573 return std::ranges::all_of(filters_, [&k](const auto& filter) {
574 return filter(k);
575 });
576 });
577 }

606 const auto& params{std::get<PointParams>(query_params_)};
