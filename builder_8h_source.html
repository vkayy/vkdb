392 result_type execute() {
393 switch (query_type_) {
394 case QueryType::NONE:
395 if (filters_.size() == 1) {
396 throw std::runtime_error{
397 "QueryBuilder::execute(): No query type specified."

402 case QueryType::POINT:

404 case QueryType::RANGE:

567 [[nodiscard]] result_type get_filtered_range() const {
568 if (query_type_ == QueryType::POINT) {
569 return execute_point_query();
570 }
571 const auto& params{std::get<RangeParams>(query_params_)};
572 return lsm_tree_.getRange(params.start, params.end,
573 [this](const auto& k) {
574 return std::ranges::all_of(filters_, [&k](const auto& filter) {
575 return filter(k);
576 });
577 });
578 }

606 [[nodiscard]] result_type execute_point_query() const {
607 const auto& params{std::get<PointParams>(query_params_)};
608 auto value{lsm_tree_.get(params.key)};
